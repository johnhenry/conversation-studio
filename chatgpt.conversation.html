'You are doing amazing!',
'Keep up the great work!',
'Believe in yourself, you got this!'

let messageIndex = 0;

const canvas = document.getElementById('messageCanvas');
const context = canvas.getContext('2d');

function displayMessage() {
// Clear the canvas
context.clearRect(0, 0, canvas.width, canvas.height);

// Set text properties
context.font = '20px Arial';
context.fillStyle = 'black';
context.textAlign = 'center';
context.textBaseline = 'middle';

// Display the current message
context.fillText(messages[messageIndex], canvas.width / 2, canvas.height / 2);
}

document.getElementById('nextMessageButton').addEventListener('click', () => {
// Cycle to the next message
messageIndex = (messageIndex + 1) % messages.length;
displayMessage();
});
